fn longest_consecutive(nums: Vec<i32>) -> i32 {
    if nums.is_empty() {
        return 0;
    }

    let nums_set: HashSet<i32> = nums.into_iter().collect();
    let mut longest = 0;

    for &num in nums_set.iter() {
        if !nums_set.contains(&(num - 1)) {
            let mut curr_num = num;
            let mut curr_length = 1;

            while nums_set.contains(&(curr_num + 1)) {
                curr_num += 1;
                curr_length += 1;
            }

            longest = longest.max(curr_length);
        }
    }

    longest
}

        }

impl Solution {
    pub fn get_common(nums1: Vec<i32>, nums2: Vec<i32>) -> i32 {
        let (mut left, mut right) = (0, 0);
        let man_len = std::cmp::max(nums1.len(), nums2.len());

        while left < man_len {
            match (nums1.get(left), nums2.get(right)) {
                (Some(v1), Some(v2)) => {
                    if v1 == v2 {
                        return *v1;
                    } else if v1 < v2 {
                        left += 1;
                    } else {
                        right += 1;
                    }
                },
                (Some(_v1), None) => break,
                (None, Some(_v2)) => break,
                (None, None) => break
            }
        }

        -1
    }
}

        let mut pre_char = 'a';

        for (i, c) in s.chars().enumerate() {
            if c == '1' {

    }
